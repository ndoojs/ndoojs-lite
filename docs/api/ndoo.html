// alias _stor
var _stor = ndoo.storage;
// set abc vlaue 1
_stor('abc', 1); // 1
// set abc value 2 failed
_stor('abc', 2); // false
// set abc value 2
_stor('abc', 2, _stor.REWRITE); // 2
// delete abc
_stor('abc', null, _stor.DESTROY); // true
